105 {
106  qDebug() << Q_FUNC_INFO << "idProcessus =" << idProcessus;
107  if (idProcessus <= 0) {

111 
112  listeDesEtapes.clear();

114 
115  db = BaseDeDonnees::getDatabase();
116  if (!db.isOpen() && !db.open()) {
117  qCritical() << "Impossible d'ouvrir la BDD :" << db.lastError().text();
118  labelEtatRequete->setText("Erreur BDD : " + db.lastError().text());
119  return;

122  QSqlQuery query(db);
123  query.prepare(R"(

138  listeDesEtapes.append(e);
139  }
140 
141  qDebug() << "Étapes chargées :" << listeDesEtapes.size();
142 
143  if (listeDesEtapes.isEmpty()) {
144  labelEtatRequete->setText("Aucune étape trouvée !");
145  } else {
146  etapeActuelIndex = recupererIndexDerniereEtape(idProcessusActuel);

149  }

371 void FenetreEtapes::traiterTrameRecue(const QString &trame)
372 {
373  qDebug() << Q_FUNC_INFO << "Trame reçue:" << trame;
374 
375  // Envoyer acquittement
376  if (!trame.contains('A')) {
377  if (communication) {
378  communication->envoyerTrame("$A%");
379  qDebug() << "Trame d'acquittement envoyée: $A%";
380  }
381  }
382 
383  if (trame == "$V") {
384  if (!attenteConfirmationPiochage) {
385  chargerEtapeSuivante();
386  } else {
387  qDebug() << "Trame $V ignorée car attente du $C après $E.";
388  }
389  }
390  else if (trame == "$E") {
391  afficherPopupDemandePiochage();
392  attenteConfirmationPiochage = true;
393  }
394  else if (trame == "$C") {
395  if (popupPiochage) {
396  popupPiochage->close();
397  popupPiochage->deleteLater();
398  popupPiochage = nullptr;
399  boutonEtapeSuivante->setEnabled(true);
400  }
401  attenteConfirmationPiochage = false; // Autorise $V après $C
402  }
403 }
